    \begin{recipe}
        [%
            preparationtime = {\unit[45]{min}},
            portion = {\unit[1]{Portion}},
            source = {Lyon 6/11/2021},
        ]
        {Caramelized pumpkin, onion and apple}

        \graph{
            big = ZuccaCipollaMela.JPG
        }

        \ingredients{
            1/2 & Red onion \\
            \unit[150]{g} & Pumpkin \\
            1 & Apple \\
            \unit[1]{spoon} & EVO Oil \\
            \unit[1/2]{glass} & Water \\
            \unit[1]{pinch} & Salt \\
            \unit[1]{pinch} & Black Pepper \\
            \unit[1]{pinch} & Sugar \\
        }

        \preparation{
            \begin{enumerate}
            \item Peel the onion and cut it into chunks. Heat the oil in a non-stick pan and saut\'e.
            \item Cut the pumpkin into chunks. When the onion begins to turn transparent add the pumpkin pieces to the pan. Sauté for a while, then add a little water and cover. Cook over medium heat for 15 minutes.
            \item Cut the apple into chunks and add it to the pan. Continue cooking, possibly adding water if the vegetables risk burning.
            \item Add salt, pepper, sugar and a little more water. Stir and continue cooking until the water is absorbed.
            \item Caramelize the mixture and serve with a garnish of balsamic vinegar and, if desired, one or two sliced kiwis.
            \end{enumerate}
        }
    \end{recipe}

    \begin{recipe}
        [%
            preparationtime = {\unit[45]{min}},
            portion = {\unit[1]{Portion}},
            source = {Lyon 10/11/2021},
        ]
        {Autumnal `Caponata'}

        \graph{
            big = CaponataAutunnale.JPG
        }

        \ingredients{
            \unit[2]{cloves} & Garlic \\
            \unit[100]{g} & Pumpkin \\
            1 & Potato \\
            1 & Tomato\\
            1/2 & Zucchini \\
            \unit[20]{g} & Green olives in brine \\
            \unit[10]{g} & Capers in brine \\
            \unit[2]{spoons} & EVO Oil \\
            \unit[1/2]{glass} & Water \\
            \unit[1]{spoon} & White wine vinegar \\
            \unit[1]{pinch} & Salt \\
            \unit[1]{pinch} & Chili \\
            \unit[1]{chunk} & Lemon Zest\\
        }

        \preparation{
            \begin{enumerate}
            \item Peel one of the two cloves of garlic and cut it into small pieces. Heat the oil in a non-stick pan and sauté briefly.
            \item Cut the pumpkin and potatoes into large pieces and add them to the pan. Sauté for about a minute, then add a little water and cover. Cook over medium heat for 10 minutes.
            \item Cut the courgette into large pieces and add it to the pan. Continue cooking for another 10 minutes, possibly adding water if the vegetables risk burning. Add the pinch of salt.
            \item Cut the tomato into large pieces and add it to the pan together with the olives and capers.
            \item Chop the lemon zest and cut the second clove of garlic into small pieces. Add both to the pan together with the sprinkling of chilli and a spoonful of vinegar.
            \item Cook for a little more until the vegetables begin to fall apart.
            \item Serve possibly accompanying with bread or a flatbread. If you want, you can add a small tasting of cheeses with honey or jams to the dish.
            \end{enumerate}
        }
    \end{recipe}
