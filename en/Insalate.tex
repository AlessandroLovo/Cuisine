    \begin{recipe}
        [%
            preparationtime = {\unit[10]{min}},
            portion = {\unit[1]{Portion}},
            source = {Lyon 29/11/2021},
        ]
        {Danish Salad}

        \graph{
            big = InsalataDanese.JPG
        }

        \ingredients{
            1/8 & Onion \\
            \unit[100]{g} & Smoked Herring \\
            \unit[150]{g} & Lamb's lettuce \\
            \unit[10]{leaves} & Basil \\
            10-15 & Green olives \\
            \unit[2]{spoons} & Capers in brine \\
            \unit[1.5]{spoons} & EVO Oil \\
            \unit[1]{tea spoon} & Honey \\
            \unit[1]{sprinkle} & Dill \\
        }

        \preparation{
            \begin{enumerate}
            \item Peel the onion and cut it into strips.
            \item Wash and centrifuge the lamb's lettuce and basil.
            \item Cut the smoked herring into large pieces.
            \item In a bowl or deep plate mix the lamb's lettuce, basil, onion, olives, capers and herring. In a small cup, mix the honey with the oil and the dill, then dress the salad.
            \end{enumerate}
        }
    \end{recipe}
